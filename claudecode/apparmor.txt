  network inet6 stream,

  /init ix,
  /bin/** ix,

  /package/** ix,

  # Access to options.json and other files within /data
  /data/** rw,

  # Access to mapped directories
  /homeassistant/** rw,
  /ssl/** r,
  /share/** rw,
  /config/** rw,

  # Claude Code and Node.js
  /usr/lib/node_modules/** ix,
  /root/** rwk,
  /root/.claude/** rwk,
  /root/.local/** rwk,
  /root/.npm/** rwk,

  /usr/bin/ttyd ix,

  /usr/lib/python*/** ix,
  /usr/bin/python* ix,

  /usr/bin/git ix,
  /usr/libexec/git-core/** ix,

  /proc/*/fd/** rw,

  /sys/kernel/** r,
